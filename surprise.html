<script>

const cards = document.querySelectorAll(".card1");

let count = 0;

cards.forEach((card)=>{

    const img1 = card.querySelector(".img1");

    const img2 = card.querySelector(".img2");

    const img3 = card.querySelector(".img3");

    card.addEventListener("click",()=>{

        img1.style.display = "none";

        img2.style.display = "block";

        setTimeout(()=>{

            img2.style.transform = "translateX(200px)";

            img2.style.transition = "1s";

            img2.style.display = "none";

            img3.style.display = "block";

        },1000);

        count++;

        if(count >= cards.length){

            setTimeout(()=>{

                document.querySelector(".card").style.display = "none";

                document.querySelector(".img4").style.display = "block";

            },2000);

        }

    });

});
